int parse_address(const char *str, Address *addr) {
    char ip_str[INET_ADDRSTRLEN];
    int port;
    
    if (sscanf(str, "%15[^:]:%d", ip_str, &port) != 2) {
        return -1;
    }
    
    if (inet_pton(AF_INET, ip_str, &addr->ip) != 1) {
        return -1;
    }
    
    addr->port = (uint16_t)port;
    return 0;
}
